新增用户

作者: caishaodong

请求URL

/user/save POST

请求体 application/json

{
	"log":{},
	"id":"long",
	"name":"string //姓名【必须】",
	"mobile":"string //手机号【必须】",
	"email":"string //邮箱",
	"password":"string //密码【必须】",
	"age":"int //年龄",
	"address":"string //住址",
	"status":"int //状态（0：正常，1：注销，2：冻结）",
	"isDeleted":"int //是否删除（0：正常，1：删除）",
	"gmtCreate":"date //创建时间",
	"gmtModified":"date //修改时间"
}

返回结果

{
	"code":"int",
	"errorMsg":"string",
	"data":{}
}
